def exclusiveTime(self, n: int, logs: List[str]) -> List[int]:
    res, stack, prev = [0] * n, [], 0
    for log in logs:
        i, op, ti = log.split(':')
        ti = int(ti) if op=='start' else int(ti)+1
        if stack:
            res[stack[-1]] += ti - prev
        if op == 'start':
            stack.append(int(i))
        else:
            stack.pop()
        prev = ti
    return res
